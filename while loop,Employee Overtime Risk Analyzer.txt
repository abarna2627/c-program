#include <stdio.h>

int main() {
    int n;
    
  
    scanf("%d", &n);
    
    int hours[n];
    
   
    for(int i = 0; i < n; i++) {
        scanf("%d", &hours[i]);
    }
    
    int i = 0;
    int totalOvertime = 0;
    int burnoutDays = 0;
    
    
    while(i < n) {
        totalOvertime += hours[i];
        
        if(hours[i] > 4) {
            burnoutDays++;
        }
        
        i++;
    }
    
   
    printf("Total Overtime: %d\n", totalOvertime);
    printf("Burnout Days: %d\n", burnoutDays);
    
    return 0;
}
