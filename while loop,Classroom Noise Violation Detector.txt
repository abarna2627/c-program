#include <stdio.h>
int main(){
int n;
scanf("%d",&n);
int i =0;
int Noise;
int Violations = 0;
int Currentstreak =0;
int ViolationStreak = 0;
while ( i < n) {
    scanf("%d",&Noise);
    if (Noise > 70) {
        Violations++;
        Currentstreak++;

        if (Currentstreak > ViolationStreak) {
            ViolationStreak = Currentstreak;
        }
    }
    else {
        Currentstreak =0;
    }
    i++;
}
printf("Noise Violations: %d\n",Violations);
printf("Longest Violation Streak: %d\n", ViolationStreak);
return 0;
}
