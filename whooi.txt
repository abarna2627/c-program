#include <stdio.h>

int main() {
    int maxWeight, n;
    
    
    scanf("%d", &maxWeight);
    
   
    scanf("%d", &n);
    
    int weights[n];
    
    for(int i = 0; i < n; i++) {
        scanf("%d", &weights[i]);
    }
    
    int i = 0;
    int currentWeight = 0;
    int peopleEntered = 0;
    int overload = 0;
    
   
    while(i < n) {
        if(currentWeight + weights[i] <= maxWeight) {
            currentWeight += weights[i];
            peopleEntered++;
        } else {
            overload = 1; 
            break;
        }
        i++;
    }
    
   
    printf("People Entered: %d\n", peopleEntered);
    
    if(overload)
        printf("Overload Status: Yes\n");
    else
        printf("Overload Status: No\n");
    
    return 0;
}
