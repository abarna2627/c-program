#include <stdio.h>

int main() {
    int atmCash, n;
    
   
    scanf("%d", &atmCash);
  
    scanf("%d", &n);
    
    int withdrawals[n];
    
   
    for(int i = 0; i < n; i++) {
        scanf("%d", &withdrawals[i]);
    }
    
    int i = 0;
    int successful = 0;
    
   
    while(i < n) {
        if(atmCash >= withdrawals[i]) {
            atmCash -= withdrawals[i];
            successful++;
        } else {
            break;  
        }
        i++;
    }
    
  
    printf("Successful Withdrawals: %d\n", successful);
    printf("Remaining Cash: %d\n", atmCash);
    
    return 0;
}
