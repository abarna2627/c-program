#include <stdio.h>

int main() {
    int N;
    scanf("%d", &N);

    int i = 0;
    int order;
    int Successful = 0;
    int Cancelled = 0;
    int Status;
    while(i < N){
        scanf("%d",&order);
        if(order == 1){
            Successful++;
            
        }
    
    else if(order == 0){
        Cancelled++;
    }
      
    i++;
    }
    printf("Successful: %d\n",Successful);
    printf("Cancelled: %d\n",Cancelled);
    if(Cancelled > Successful){
        printf("Status: Risk\n");
    } else {
        printf("Status: Safe\n");
    }
    

    return 0;
}
