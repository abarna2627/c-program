#include <stdio.h>

int main() {
    int capacity, n;
    
   
    scanf("%d", &capacity);
    
   
    scanf("%d", &n);
    
    int load[n];
    
    
    for(int i = 0; i < n; i++) {
        scanf("%d", &load[i]);
    }
    
    int i = 0;
    int safeHours = 0;
    int failureCount = 0;
    
   
    while(i < n) {
        if(load[i] <= capacity) {
            safeHours++;
        } else {
            failureCount++;
        }
        i++;
    }
    
   
    printf("Safe Hours: %d\n", safeHours);
    printf("Failure Count: %d\n", failureCount);
    
    return 0;
}
