#include <stdio.h>

int main() {
    int n;
    
    scanf("%d", &n);
    
    int transactions[n];
    
   for(int i = 0; i < n; i++) {
        scanf("%d", &transactions[i]);
    }
    
    int i = 0;
    int consecutiveHigh = 0;
    int highValueCount = 0;
    int fraudAttempt = 0;
    
   
    while(i < n) {
        if(transactions[i] >= 50000) {
            highValueCount++;
            consecutiveHigh++;
          
            if(consecutiveHigh == 3 && fraudAttempt == 0) {
                fraudAttempt = i + 1; 
            }
        } else {
            consecutiveHigh = 0;
        }
        i++;
    }
    
  
    if(fraudAttempt > 0) {
        printf("Fraud Triggered At Attempt: %d\n", fraudAttempt);
    } else {
        printf("Fraud Triggered At Attempt: Not Triggered\n");
    }
    
    printf("High-Value Transactions: %d\n", highValueCount);
    
    return 0;
}
