#include <stdio.h>

int main() {
    int N;
    scanf("%d", &N);
    
    int units;
    int totalunits = 0;
    int spikehours = 0;
    
   
    for(int i = 0; i < N; i++) {
        scanf("%d", &units);
        totalunits += units;
        
      
        if(units > 5) {
          spikehours ++;
        }
    }
    
   
    printf("Total Units: %d\n", totalunits);
    printf("Spike Hours: %d\n", spikehours);
    
    return 0;
}
