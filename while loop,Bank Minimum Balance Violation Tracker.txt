#include <stdio.h>

int main() {
    int N;
    scanf("%d", &N);

    int initialBalance;
    scanf("%d", &initialBalance);

    int FinalBalance = initialBalance;
    int LowBalanceDays = 0;
    int i = 0;
    int Transaction;

    while (i < N) {
        scanf("%d", &Transaction);
        FinalBalance += Transaction;

       
        if (FinalBalance < 2000 && i != N - 1) {
            LowBalanceDays++;
        }

        i++;
    }

    printf("Final Balance: %d\n", FinalBalance);
    printf("Low Balance Days: %d\n", LowBalanceDays);

    return 0;
}
