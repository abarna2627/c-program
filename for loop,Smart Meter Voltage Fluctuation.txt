#include <stdio.h>

int main() {
    int N;
    scanf("%d", &N);

    int voltage_reading;
    int Low_Voltage_Events = 0;
    int Minimum_Voltage;

    for(int i = 0; i < N; i++) {
        scanf("%d", &voltage_reading);

        if(i == 0) {
           
            Minimum_Voltage = voltage_reading;
        } else if(voltage_reading < Minimum_Voltage) {
            Minimum_Voltage = voltage_reading;
        }

        if(voltage_reading < 210) {
            Low_Voltage_Events++;
        }
    }

    printf("Minimum Voltage: %d\n", Minimum_Voltage);
    printf("Low Voltage Events: %d\n", Low_Voltage_Events);

    return 0;
}
