#include <stdio.h>

int main() {
   int N;
   scanf("%d",&N);
   int per_days;
   int Highest_Price = 0,High_Price_Days = 0;
   for(int i = 0; i<N; i++){
       scanf("%d",&per_days);
       if(per_days > Highest_Price) {
       Highest_Price = per_days;
       }
       if(per_days > 100) {
          High_Price_Days++;
       }
       }
       printf("Highest price:%d\n",Highest_Price);
       printf("High Price Days:%d\n",High_Price_Days);
    return 0;
}
