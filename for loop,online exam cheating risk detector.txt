#include <stdio.h>

int main() {
    int N;
    scanf("%d", &N);
    
    int scores[N];
    int totalRisk = 0;
    int highRiskCount = 0;
    
   
    for(int i = 0; i < N; i++) {
        scanf("%d", &scores[i]);
        totalRisk += scores[i];
        
      
        if(scores[i] > 50) {
            highRiskCount++;
        }
    }
    
   
    printf("Total Risk: %d\n", totalRisk);
    printf("High Risk Sessions: %d\n", highRiskCount);
    
    return 0;
}
