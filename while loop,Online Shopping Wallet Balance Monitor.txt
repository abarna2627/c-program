#include <stdio.h>

int main() {
    int walletBalance, n;
    
    scanf("%d", &walletBalance);
    
    scanf("%d", &n);
    
    int purchases[n];
    
    for(int i = 0; i < n; i++) {
        scanf("%d", &purchases[i]);
    }
    
    int i = 0;
    int successfulPurchases = 0;
    
    while(i < n) {
        if(walletBalance >= purchases[i]) {
            walletBalance -= purchases[i];
            successfulPurchases++;
        } else {
            break;  
        }
        i++;
    }
    
    
    printf("Successful Purchases: %d\n", successfulPurchases);
    printf("Final Balance: %d\n", walletBalance);
    
    return 0;
}
