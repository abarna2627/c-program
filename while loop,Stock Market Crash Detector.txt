#include <stdio.h>

int main() {
    int n;
    
 
    scanf("%d", &n);
    
    int index[n];
    
   
    for(int i = 0; i < n; i++) {
        scanf("%d", &index[i]);
    }
    
    int i = 1;
    int consecutiveDrops = 0;
    int totalDrops = 0;
    int crashDay = 0;
    
 
    while(i < n) {
        if(index[i] < index[i - 1]) {
            totalDrops++;
            consecutiveDrops++;
        
            if(consecutiveDrops == 3 && crashDay == 0) {
                crashDay = i + 1; 
            }
        } else {
            consecutiveDrops = 0;
        }
        
        i++;
    }
    
   
    printf("Crash Day: %d\n", crashDay);
    printf("Total Drops: %d\n", totalDrops);
    
    return 0;
}
